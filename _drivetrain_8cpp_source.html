196bool Drivetrain::Move(double distance, double speed)
197{
198 m_movePIDController.SetSetpoint(distance);
199 double output = m_movePIDController.Calculate(GetEncoderAverage() * m_movePIDDirection);
200 output = clamp(output, -speed, speed);
201 Drive(output * speed, 0);
202 return m_movePIDController.AtSetpoint();
203}

246bool Drivetrain::MoveTo(double x, double y, double speed, double turnSpeed)
247{
248 double targetX = x - m_currentX;
249 double targetY = y - m_currentY;
250
251 double angle = GetAbsoluteAngle(targetX, targetY);
252
253 if (Turn(angle, turnSpeed))
254 {
255 double distance = sqrt(pow(targetX, 2) + pow(targetY, 2));
256
257 if (Move(distance, speed))
258 {
259 ResetEncoders();
260 m_currentX = x;
261 m_currentY = y;
262 return true;
263 }
264
265 return false;
266 }
267
268 return false;
269}

348double Drivetrain::GetAbsoluteAngle(double x, double y)
349{
350 float relAngle = atan(y / (x == 0 ? 0.01 : x));
351
352 if (x < 0)
353 relAngle += M_PI;
354
355 else if (y < 0)
356 relAngle += 2 * M_PI;
357
358 return (relAngle)*180 / M_PI;
359}
