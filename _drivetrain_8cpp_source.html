158bool Drivetrain::Move(double distance, double speed)
159{
160 m_movePIDController.SetSetpoint(distance);
161 m_movePIDController.SetTolerance(k_movePIDTolerance);
162 double output = m_movePIDController.Calculate(-GetEncoderAverage());
163 output = clamp(output, -speed, speed);

178bool Drivetrain::MoveTo(double x, double y, double speed, double turnSpeed)
179{
180 double targetX = x - m_currentX;
181 double targetY = y - m_currentY;
182
183 double angle = GetAbsoluteAngle(targetX, targetY);
184
185 if (Turn(angle, turnSpeed))
186 {
187 double distance = sqrt(pow(targetX, 2) + pow(targetY, 2));
188
189 if (Move(distance, speed))
190 {
191 ResetEncoders();
192 m_currentX = x;
193 m_currentY = y;
194 return true;
195 }
196
197 return false;
198 }
199
200 return false;
201}

285double Drivetrain::GetAbsoluteAngle(double x, double y)
286{
287 float relAngle = atan(y / (x == 0 ? 0.01 : x));
288
289 if (x < 0)
290 relAngle += M_PI;
291
292 else if (y < 0)
293 relAngle += 2 * M_PI;
294
295 return (relAngle)*180 / M_PI;
296}
