119void EncoderSubsystem::SetMotor(double speed)
120{
121
122 if (m_motorCount > 1)
123 {
124 throw std::invalid_argument("EncoderSubsystem: SetMotor() does not support multiple motors.");
125 }
126
127 if (m_limitSafetyActive)
128 {
129 if (GetUpperLimit())
130 {
131 speed = fmin(speed, 0);
132 }
133 else if (GetLowerLimit())
134 {
135 speed = fmax(speed, 0);
136 }
137 }
138
139 if (m_positionSafetyActive)
140 {
141 if (GetPosition() >= m_maxPosition)
142 {
143 speed = fmin(speed, 0);
144 }
145 else if (GetPosition() <= m_minPosition)
146 {
147 speed = fmax(speed, 0);
148 }
149 }
150
151 switch (m_motorConfig)
152 {
153 case MotorConfig::kNeo:
154 case MotorConfig::kSpark:
155 m_motorSpark->Set(std::clamp(speed, -m_maxSpeed, m_maxSpeed));
156 break;
157
158 case MotorConfig::kVictorCAN:
159 m_motorVictorCAN->Set(VictorSPXControlMode::PercentOutput, std::clamp(speed, -m_maxSpeed, m_maxSpeed));
160 break;
161
162 case MotorConfig::kVictorPWM:
163 m_motorVictorPWM->Set(std::clamp(speed, -m_maxSpeed, m_maxSpeed));
164 break;
165 }
166}

168void EncoderSubsystem::SetMotors(vector<double> speeds)
169{
170
171 if (m_motorCount != speeds.size())
172 {
173 throw std::invalid_argument("EncoderSubsystem: Speed vector must be the same size as the number of motors.");
174 }
175 else if (m_motorCount <= 1)
176 {
177 throw std::invalid_argument("EncoderSubsystem: This subsystem has only one motor. Use SetMotor() instead.");
178 }
179
180 for (int i = 0; i < speeds.size(); i++)
181 {
182 double speed = speeds[i];
183 if (m_limitSafetyActive)
184 {
185 if (GetUpperLimit())
186 {
187 speed = fmin(speed, 0);
188 }
189 else if (GetLowerLimit())
190 {
191 speed = fmax(speed, 0);
192 }
193 }
194
195 if (m_positionSafetyActive)
196 {
197 if (GetPosition() >= m_maxPosition)
198 {
199 speeds[i] = fmin(speed, 0);
200 }
201 else if (GetPosition() <= m_minPosition)
202 {
203 speeds[i] = fmax(speed, 0);
204 }
205 }
206
207 switch (m_motorConfig)
208 {
209 case MotorConfig::kNeo:
210 case MotorConfig::kSpark:
211 m_motorSpark->Set(std::clamp(speed, -m_maxSpeed, m_maxSpeed));
212 break;
213
214 case MotorConfig::kVictorCAN:
215 m_motorVictorCAN->Set(VictorSPXControlMode::PercentOutput, std::clamp(speed, -m_maxSpeed, m_maxSpeed));
216 break;
217
218 case MotorConfig::kVictorPWM:
219 m_motorVictorPWM->Set(std::clamp(speed, -m_maxSpeed, m_maxSpeed));
220 break;
221 }
222 }
223}
